\begin{code}
  data `MMT : Set where
    `cut `μ `μ∼ : (A : Ty) → `MMT
    `λ `λ∼ : (A : Ty) → `MMT
    `⟨-,-⟩ `μ⟨-,-⟩ : (rA sB : Ann × Ty) → `MMT

  MMT : System
  MMT = `MMT ▹ λ where
    (`cut A)  → ⟨ []ᶜ `⊢ trm A ⟩ `✴ ⟨ []ᶜ `⊢ cot A ⟩  `⊆ com
    (`μ A)    → ⟨ [ 1# , cot A ]ᶜ `⊢ com ⟩            `⊆ trm A
    (`μ∼ A)   → ⟨ [ 1# , trm A ]ᶜ `⊢ com ⟩            `⊆ cot A
    (`λ A)    → ⟨ []ᶜ `⊢ cot A ⟩                      `⊆ trm (A ^⊥)
    (`λ∼ A)   → ⟨ []ᶜ `⊢ trm A ⟩                      `⊆ cot (A ^⊥)
    (`⟨-,-⟩ rA@(r , A) sB@(s , B)) →
      r `· ⟨ []ᶜ `⊢ cot A ⟩ `✴ s `· ⟨ []ᶜ `⊢ cot B ⟩  `⊆ cot (rA ⅋ sB)
    (`μ⟨-,-⟩ rA@(r , A) sB@(s , B)) →
      ⟨ [ r , cot A ]ᶜ ++ᶜ [ s , cot B ]ᶜ `⊢ com ⟩    `⊆ trm (rA ⅋ sB)
\end{code}
%</MMT>
